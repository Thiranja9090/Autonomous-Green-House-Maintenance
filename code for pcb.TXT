#include <WiFi.h>
#include <PubSubClient.h>
#include <DHT.h>
#include <Adafruit_SSD1306.h>
#include <Wire.h>

// Wi-Fi credentials
const char* ssid = "Your_SSID";
const char* password = "Your_PASSWORD";

// MQTT credentials
const char* mqttServer = "mqtt.thingspeak.com";  // Change this to your MQTT broker address
const int mqttPort = 1883;
const char* mqttUser = "Your_MQTT_USERNAME";
const char* mqttPassword = "Your_MQTT_PASSWORD";
const char* mqttTopic = "Your/ThingSpeak/Channel";

// DHT11 Sensors
#define DHTPIN1 18  // Greenhouse 1 DHT11
#define DHTPIN2 5   // Greenhouse 2 DHT11
#define DHTTYPE DHT11

DHT dht1(DHTPIN1, DHTTYPE);
DHT dht2(DHTPIN2, DHTTYPE);

// LDR Sensor (Greenhouse 1, 3.3V)
#define LDR_PIN 34  // Analog pin for LDR (3.3V)

// Soil Moisture Sensor (Greenhouse 2)
#define SOIL_MOISTURE_PIN 34  // Analog pin for soil moisture sensor (capacitive)

#define MIST_PIN 32  // Digital pin for mist maker control (5V)

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// MQTT Client
WiFiClient espClient;
PubSubClient client(espClient);

// Calibration Thresholds
int LDR_THRESHOLD = 768;    // LDR threshold for light: 75% of 1023 (maximum ADC value)
int SOIL_DRY_THRESHOLD = 45;    // Soil moisture threshold for "High Dryness"
int SOIL_WET_THRESHOLD = 70;    // Soil moisture threshold for "Too much water"
int HUMIDITY_THRESHOLD = 50;    // Humidity threshold for turning on mist maker

void setup() {
  Serial.begin(115200);

  // Initialize DHT sensors
  dht1.begin();
  dht2.begin();

  // Initialize Wi-Fi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }
  Serial.println("Connected to WiFi");

  // Initialize MQTT
  client.setServer(mqttServer, mqttPort);
  client.setCallback(mqttCallback);

  // Initialize Mist Maker control pin
  pinMode(MIST_PIN, OUTPUT);
  digitalWrite(MIST_PIN, LOW); // Ensure the mist maker is off initially

  // Initialize OLED display
  Wire.begin();
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED initialization failed!");
    while (1);
  }
  display.display();
  delay(2000);
  display.clearDisplay();
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();

  // Read DHT sensor data
  float temp1 = dht1.readTemperature();
  float hum1 = dht1.readHumidity();
  float temp2 = dht2.readTemperature();
  float hum2 = dht2.readHumidity();

  // Read LDR (Light sensor) and soil moisture sensor
  int ldrValue = analogRead(LDR_PIN);
  int soilMoisture = analogRead(SOIL_MOISTURE_PIN);

  // Handle sensor errors
  if (isnan(temp1) || isnan(hum1)) {
    temp1 = hum1 = 0.0;
  }
  if (isnan(temp2) || isnan(hum2)) {
    temp2 = hum2 = 0.0;
  }

  // Control Mist Maker based on humidity (Greenhouse 1)
  if (hum1 < HUMIDITY_THRESHOLD) {  // If humidity in GH1 is below 50%
    digitalWrite(MIST_PIN, HIGH);  // Turn on the mist maker
  } else {
    digitalWrite(MIST_PIN, LOW);  // Turn off the mist maker
  }

  // Check LDR value and display message if light is insufficient
  int ldrPercentage = map(ldrValue, 0, 1023, 0, 100);  // Convert LDR to percentage
  if (ldrPercentage < 75) {  // If LDR value is less than 75% of max
    displayMessage("Light not enough!");
  } else {
    displayMessage("");  // Clear the message if light is sufficient
  }

  // Check soil moisture value and display messages accordingly
  int soilMoisturePercentage = map(soilMoisture, 0, 900, 0, 100);  // Convert soil moisture to percentage

  if (soilMoisturePercentage < SOIL_DRY_THRESHOLD) {  // If soil moisture is below 45%
    displayMessage("High Dryness");
  } else if (soilMoisturePercentage > SOIL_WET_THRESHOLD) {  // If soil moisture is above 70%
    displayMessage("Too much water!");
  } else {
    displayMessage("");  // No message if soil moisture is within the desired range
  }

  // Prepare data for MQTT
  String payload = String("GH1_Temp:") + temp1 + ",GH1_Hum:" + hum1 + ",GH2_Temp:" + temp2 + ",GH2_Hum:" + hum2 + ",Soil_Moisture:" + soilMoisturePercentage + ",LDR:" + ldrPercentage;

  // Publish data to MQTT
  client.publish(mqttTopic, payload.c_str());

  // Display data on OLED
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);

  // GH1 and GH2 titles (yellow area at the top)
  display.setCursor(40, 2);
  display.print("GH1");
  display.setCursor(90, 2);
  display.print("GH2");

  // Display temperature and humidity readings
  display.setCursor(0, 16);
  display.print("Temp: ");
  display.print(temp1);
  display.print("C");
  display.setCursor(40, 16);
  display.print("Temp: ");
  display.print(temp2);
  display.print("C");

  display.setCursor(0, 28);
  display.print("Hum: ");
  display.print(hum1);
  display.print("%");
  display.setCursor(40, 28);
  display.print("Hum: ");
  display.print(hum2);
  display.print("%");

  display.setCursor(0, 40);
  display.print("LDR: ");
  display.print(ldrPercentage);
  display.print("%");
  display.setCursor(40, 40);
  display.print("Soil Moisture: ");
  display.print(soilMoisturePercentage);
  display.print("%");

  display.display();

  // Wait before next reading
  delay(2000);
}

void mqttCallback(char* topic, byte* payload, unsigned int length) {
  // Handle incoming messages (if needed)
}

void reconnect() {
  // Loop until we're reconnected to MQTT broker
  while (!client.connected()) {
    if (client.connect("ESP32Client", mqttUser, mqttPassword)) {
      // Subscribe to a topic (if needed)
      client.subscribe("your/topic");
    } else {
      delay(5000);
    }
  }
}

// Display message on OLED (used for LDR and soil moisture messages)
void displayMessage(String message) {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);
  display.setCursor(0, 0);
  display.print(message);
  display.display();
}
